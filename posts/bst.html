Binary Search



Time Complexity: O(N) Auxiliary Space: O(log(N)) for recursive stack space for the height of the tree

Given a sorted array, create a BST: 
            
              class Node
                attr_accessor :value, :left, :right

                def initialize(value, left=nil, right=nil)
                  @value = value
                  @left = left
                  @right = right
                end

                def self.preorder(node)
                  return unless node

                  puts node.value
                  preorder(node.left)
                  preorder(node.right)
                end

                def self.search(root, value)
                  return "not found" if root.nil?

                  return root if root.value == value

                  if value < root.value
                    search(root.left, value)
                  else
                    search(root.right, value)
                  end
                end
              end

              class SortedArray
                def self.to_bst(arr=self)
                  return if arr.empty?

                  mid = arr.length / 2

                  root = Node.new(arr[mid])
                  root.left = to_bst(arr[0...mid])
                  root.right = to_bst(arr[mid + 1...arr.length])

                  root
                end
              end
