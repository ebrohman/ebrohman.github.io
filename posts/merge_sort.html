Time Complexity: O(NlogN) Auxiliary Space: O(N) for recursive stack space 
            
              def mergesort(a)
                if a.length <= 1
                  a
                else
                  mid = a.length / 2

                  left = mergesort(a.slice(0, mid))
                  right = mergesort(a.slice(mid, a.length))
                  merge(left, right)
                end
              end


              def merge(left_ary, right_ary)
                ret = []

                while !left_ary.empty? && ! right_ary.empty? do
                  if left_ary[0] < right_ary[0]
                    ret.push left_ary.shift
                  else
                    ret.push right_ary.shift
                  end
                end

                ret.concat(left_ary).concat(right_ary)
              end
